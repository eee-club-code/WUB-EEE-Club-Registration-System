  (function() {
    const STORAGE_KEY = 'wub_eee_registrations';
    const ID_KEY = 'wub_eee_last_id';
    const year = new Date().getFullYear();

    // ---- helpers ----
    function getLastSerial() {
      const stored = localStorage.getItem(ID_KEY);
      return stored ? parseInt(stored, 10) : 0;
    }
    function setLastSerial(serial) {
      localStorage.setItem(ID_KEY, String(serial));
    }
    function generateClubId(serial) {
      return `WUB-EEE-${year}-${String(serial).padStart(4, '0')}`;
    }
    function getRegistrations() {
      const data = localStorage.getItem(STORAGE_KEY);
      return data ? JSON.parse(data) : [];
    }
    function saveRegistrations(records) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(records));
    }
    function escapeHtml(text) {
      if (!text) return '';
      const map = { '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#039;' };
      return String(text).replace(/[&<>"']/g, m => map[m]);
    }

    // ---- Email simulation ----
    function sendConfirmationEmail(record) {
      return new Promise((resolve) => {
        setTimeout(() => {
          const emailContent = `
            Subject: WUB EEE Club Registration Confirmation

            Dear ${record.fullName},

            Congratulations! Your registration for the WUB EEE Club has been confirmed.

            Club ID: ${record.clubId}
            Student ID: ${record.studentId}
            Department: EEE
            Semester: ${record.semester}
            Email: ${record.email}
            Payment: ${record.paymentStatus} (TrxID: ${record.trxid || 'N/A'})

            Thank you for joining the WUB EEE Club!

            Best regards,
            WUB EEE Club Team
            EEE Club Office, World University of Bangladesh
            Email: eee-club@eee.wub.edu.bd
          `;
          
          console.log('📧 Confirmation email sent to:', record.email);
          console.log('📧 From: eee-club@eee.wub.edu.bd');
          console.log('📧 Content:', emailContent);
          
          showToast(`✅ Confirmation email sent to ${record.email}`);
          resolve(true);
        }, 1000);
      });
    }

    // ---- Toast notification ----
    function showToast(message) {
      const existing = document.querySelector('.notification-toast');
      if (existing) existing.remove();
      
      const toast = document.createElement('div');
      toast.className = 'notification-toast';
      toast.innerHTML = `<i class="fas fa-envelope"></i> ${message}`;
      document.body.appendChild(toast);
      
      setTimeout(() => {
        toast.classList.add('hidden');
        setTimeout(() => toast.remove(), 500);
      }, 4000);
    }

    // ---- payment state ----
    let paymentCompleted = false;
    let verifiedTrxID = '';

    function updatePaymentUI() {
      const statusEl = document.getElementById('paymentStatus');
      const trxInput = document.getElementById('trxidInput');
      const verifyBtn = document.getElementById('verifyTrxidBtn');
      const submitBtn = document.getElementById('submitBtn');
      
      if (paymentCompleted && verifiedTrxID) {
        statusEl.innerHTML = `<i class="fas fa-check-circle" style="color:#1b754b;"></i> Paid (TrxID: ${verifiedTrxID})`;
        statusEl.className = 'payment-status done';
        trxInput.disabled = true;
        verifyBtn.disabled = true;
        verifyBtn.innerHTML = '<i class="fas fa-check"></i> Verified';
        verifyBtn.style.background = '#1b754b';
        submitBtn.disabled = false;
      } else {
        statusEl.innerHTML = '<i class="fas fa-clock"></i> Pending';
        statusEl.className = 'payment-status';
        trxInput.disabled = false;
        verifyBtn.disabled = false;
        verifyBtn.innerHTML = '<i class="fas fa-check"></i> Verify';
        verifyBtn.style.background = '#1b754b';
        submitBtn.disabled = true;
      }
    }

    // Verify TrxID
    document.getElementById('verifyTrxidBtn').addEventListener('click', function() {
      const trxInput = document.getElementById('trxidInput');
      const trxid = trxInput.value.trim();
      if (!trxid) {
        alert('Please enter a bKash Transaction ID (TrxID).');
        return;
      }
      if (trxid.length < 4) {
        alert('Please enter a valid TrxID (at least 4 characters).');
        return;
      }
      this.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Verifying...';
      this.disabled = true;
      setTimeout(() => {
        verifiedTrxID = trxid;
        paymentCompleted = true;
        updatePaymentUI();
        alert(`✅ bKash payment verified!\nTrxID: ${verifiedTrxID}\nAmount: 200 TK`);
        this.innerHTML = '<i class="fas fa-check"></i> Verified';
        this.style.background = '#1b754b';
      }, 800);
    });

    // Reset payment
    document.getElementById('resetPaymentBtn').addEventListener('click', function() {
      paymentCompleted = false;
      verifiedTrxID = '';
      document.getElementById('trxidInput').value = '';
      document.getElementById('trxidInput').disabled = false;
      const verifyBtn = document.getElementById('verifyTrxidBtn');
      verifyBtn.disabled = false;
      verifyBtn.innerHTML = '<i class="fas fa-check"></i> Verify';
      verifyBtn.style.background = '#1b754b';
      document.getElementById('submitBtn').disabled = true;
      updatePaymentUI();
    });

    // ---- form submit ----
    document.getElementById('registrationForm').addEventListener('submit', async function(e) {
      e.preventDefault();

      if (!paymentCompleted || !verifiedTrxID) {
        alert('⚠️ Please enter and verify your bKash TrxID first.');
        return;
      }

      const fullName = document.getElementById('fullName').value.trim();
      const studentId = document.getElementById('studentId').value.trim();
      const email = document.getElementById('email').value.trim();
      if (!fullName || !studentId || !email) {
        alert('Please fill in all required fields: Full Name, Student ID, and Email.');
        return;
      }

      // Validate email format
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(email)) {
        alert('Please enter a valid email address.');
        return;
      }

      const skillCheckboxes = document.querySelectorAll('#skillsGroup input[type="checkbox"]');
      const skills = [];
      skillCheckboxes.forEach(cb => { if (cb.checked) skills.push(cb.value); });

      const semester = document.getElementById('semester').value;
      const phone = document.getElementById('phone').value.trim();
      const gender = document.getElementById('gender').value;
      const interest = document.getElementById('interest').value;
      const reason = document.getElementById('reason').value.trim();

      let lastSerial = getLastSerial();
      const newSerial = lastSerial + 1;
      const clubId = generateClubId(newSerial);
      setLastSerial(newSerial);

      const record = {
        fullName, studentId, department: 'EEE', semester, email, phone, gender,
        skills, interest, reason, clubId,
        paymentStatus: 'Paid (bKash)',
        paymentMethod: 'bKash',
        paymentAmount: '200 TK',
        trxid: verifiedTrxID,
        timestamp: new Date().toISOString()
      };

      const records = getRegistrations();
      const exists = records.some(r => r.studentId === studentId);
      if (exists) {
        alert('This Student ID is already registered. Please use a different ID.');
        setLastSerial(lastSerial);
        return;
      }
      records.push(record);
      saveRegistrations(records);

      // Send confirmation email
      await sendConfirmationEmail(record);
      
      // show success
      const box = document.getElementById('successBox');
      const span = document.getElementById('displayClubId');
      if (span) span.textContent = clubId;
      if (box) box.classList.add('show');

      window._lastClubId = clubId;
      window._lastFormData = record;

      renderAdminTable();
      document.getElementById('successBox').scrollIntoView({ behavior: 'smooth', block: 'center' });
      document.getElementById('submitBtn').disabled = true;
    });

    // admin table
    function renderAdminTable() {
      const records = getRegistrations();
      const tbody = document.getElementById('adminTableBody');
      if (!tbody) return;
      if (records.length === 0) {
        tbody.innerHTML = `<tr class="empty-row"><td colspan="7">No registered members yet. Be the first!</td></tr>`;
        return;
      }
      tbody.innerHTML = records.map((r, idx) => {
        const paid = r.paymentStatus && r.paymentStatus.includes('Paid');
        return `<tr>
          <td>${idx+1}</td>
          <td><strong>${escapeHtml(r.fullName || '')}</strong></td>
          <td>${escapeHtml(r.studentId || '')}</td>
          <td><code style="background:#e8f0fe; padding:2px 8px; border-radius:12px; font-weight:600;">${escapeHtml(r.clubId || '')}</code></td>
          <td><span class="${paid ? 'badge-paid' : 'badge-pending'}">${paid ? '✅ Paid' : '⏳ Pending'}</span></td>
          <td><code style="font-size:0.7rem; background:#f0f0f0; padding:2px 6px; border-radius:4px;">${escapeHtml(r.trxid || '—')}</code></td>
          <td style="font-size:0.8rem;">${escapeHtml(r.email || '—')} <span class="email-sent"><i class="fas fa-check-circle"></i></span></td>
        </tr>`;
      }).join('');
    }
